42  public class ManifestBundleImportVersionUpdaterTransformer {

44      private Map<String, String> entriesVersionUpdates = new HashMap<String, String>();

52      private static ManifestBundleImportVersionUpdaterTransformer getInstance() {
53          if (instance == null) {
54              instance = new ManifestBundleImportVersionUpdaterTransformer();
55          }
56          return instance;

59      public boolean canTransformResource(String resource) {
60          if (JarFile.MANIFEST_NAME.equalsIgnoreCase(resource)) {
61              return true;
62          }
63  
64          return false;
65      }

67      public void processResource(InputStream is) throws IOException {
68          // We just want to take the first manifest we come across as that's our
69          // project's manifest. This is the behavior
70          // now which is situational at best. Right now there is no context
71          // passed in with the processing so we cannot
72          // tell what artifact is being processed.
73          if (!manifestDiscovered) {
74              manifest = new Manifest(is);
75              manifestDiscovered = true;
76              IOUtil.close(is);
77          }
78      }

84      public void updateManifestImportsWithOverrides() throws IOException {
85          // If we didn't find a manifest, then let's create one.
86          if (manifest == null) {
87              System.out.println("New Manifest");
88              manifest = new Manifest();
89          }
90  
91          Attributes attributes = manifest.getMainAttributes();
92          StringBuilder updatedImports = new StringBuilder();
93  
94          if (entriesVersionUpdates != null) {
95              String imports = attributes.getValue(Constants.IMPORT_PACKAGE);
96              StringBuilder regexp = new StringBuilder();
97              regexp.append("([a-zA-Z0-9\\.]+?)");//symbolicname
98              regexp.append("((;[^=]*?=([^;,]|,\\s*[\\.\\]\\)0-9]*\")*?)*?)");//specialization
99              regexp.append("(,(?!\\s*[\\.\\]\\)0-9]*\")|$)");//iterator
100             Pattern pattern = Pattern
101                     .compile(regexp.toString());// ([a-z|\\.]+?)((;\\w*?=\".*?\")*)(,|$)
102             Matcher matcher = pattern.matcher(imports);
103             while (matcher.find()) {
104                 String symbolicName = matcher.group(1);
105                 Boolean foundMatch = Boolean.FALSE;
106                 updatedImports.append(symbolicName);
107                 for (Iterator i = entriesVersionUpdates.keySet().iterator(); i
108                         .hasNext();) {
109                     String key = (String) i.next();
110                     if (key.equals(symbolicName)) {
111                         String specialization = matcher.group(2);
112                         if (specialization != null) {
113                             foundMatch = Boolean.TRUE;
114                             System.out.println("Replacement matched: "
115                                     + symbolicName + "," + specialization);
116                             Pattern versionDeclarationPattern = Pattern.compile(";\\s*version\\s*=\\s*\"");
117                             Matcher versionDeclarationMatcher = versionDeclarationPattern.matcher(specialization);
118                             if (!versionDeclarationMatcher.find()) {
119                             	specialization = specialization + ";version=\""
120                                         + entriesVersionUpdates.get(key)
121                                         + "\"";
122                             } else {
123                             specialization = specialization.replaceAll(
124                                     ";\\s*version\\s*=\\s*\".*?\"",
125                                     ";version=\""
126                                             + entriesVersionUpdates.get(key)
127                                             + "\"");
128                             }
129                             updatedImports.append(specialization);
130                         }
131                     }
132                 }
133                 if (!foundMatch) {
134                     String specialization = matcher.group(2);
135                     updatedImports.append(specialization);
136 
137                 }
138                 updatedImports.append(",");
139 
140             }
141 
142         }
143         attributes.put(
144                 new Attributes.Name(Constants.IMPORT_PACKAGE),
145                 updatedImports.toString().substring(0,
146                         updatedImports.toString().length() - 1));
147 
148     }

150     public static void main(String[] args) throws IOException {
151         String[] overrides = null;
152         String basedir = null;
153         String bundleClasspath = null;
154         for (String arg : args) {
155             if (arg.startsWith("overrides=")) {
156                 overrides = arg.replaceFirst("overrides=", "").split(";");
157             } else if (arg.startsWith("basedir=")) {
158                 basedir = arg.replaceFirst("basedir=", "");
159             } else if (arg.startsWith("bundleClasspath=")) {
160                 bundleClasspath = arg.replaceFirst("bundleClasspath=", "");
161             }
162             System.out.println("Argument: " + arg);
163 
164         }
165         InputStream is;
166 
167         try {
168             is = new FileInputStream(basedir
169                     + "/src/main/resources/META-INF/MANIFEST.MF");
170             getInstance().processResource(is);
171             is.close();
172         } catch (FileNotFoundException e) {
173             // TODO Auto-generated catch block
174             e.printStackTrace();
175         }
176         getInstance().createOverrideMap(overrides);
177         getInstance().updateManifestImportsWithOverrides();
178         getInstance().addBundleClassPath(bundleClasspath);
179         getInstance().writeManifest(basedir);
180     }
181 
182     private void writeManifest(String basedir) throws IOException {
183         OutputStream os;
184         try {
185             os = new FileOutputStream(basedir
186                     + "/src/main/resources/META-INF/MANIFEST.MF");
187             manifest.write(os);
188             os.close();
189         } catch (FileNotFoundException e) {
190             // TODO Auto-generated catch block
191             e.printStackTrace();
192         }
193     }
194 
195     private void addBundleClassPath(String bundleClasspath) {
196         Attributes attributes = manifest.getMainAttributes();
197         attributes.put(new Attributes.Name(Constants.BUNDLE_CLASSPATH), "., "
198                 + bundleClasspath + ".jar");
199     }
200 
201     private void createOverrideMap(String[] overrides) {
202         for (String override : overrides) {
203             String[] mapEntry = override.split("=");
204             if (mapEntry.length == 2) {
205                 entriesVersionUpdates.put(mapEntry[0], mapEntry[1]);
206             }
207         }
208     }
