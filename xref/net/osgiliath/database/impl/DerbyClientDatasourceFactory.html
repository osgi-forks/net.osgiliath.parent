44  @Slf4j
45  public class DerbyClientDatasourceFactory implements DataSourceFactory {
46  
47  	@Override
48  	public DataSource createDataSource(Properties props) throws SQLException {
49  		ClientDataSource40 ds = new ClientDataSource40();
50  		setProperties(ds, props);
51  		return ds;
52  	}
53  
54  	private void setProperties(ClientDataSource ds, Properties properties)
55  			throws SQLException {
56  		Properties props = (Properties) properties.clone();
57  		String doStartServer = (String) props
58  				.remove(ClientConnectionConstant.AUTO_START_SERVER);
59  		if (doStartServer != null) {
60  			if (Boolean.parseBoolean(doStartServer)) {
61  				doStartServer(props);
62  			}
63  		}
64  		String databaseName = (String) props
65  				.remove(DataSourceFactory.JDBC_DATABASE_NAME);
66  		if (databaseName == null) {
67  			throw new SQLException("missing required property "
68  					+ DataSourceFactory.JDBC_DATABASE_NAME);
69  		}
70  		ds.setDatabaseName(databaseName);
71  
72  		String password = (String) props
73  				.remove(DataSourceFactory.JDBC_PASSWORD);
74  		ds.setPassword(password);
75  
76  		String user = (String) props.remove(DataSourceFactory.JDBC_USER);
77  		ds.setUser(user);
78  
79  		String createDatabase = (String) props
80  				.remove(ClientConnectionConstant.CREATE_DATABASE);
81  		ds.setCreateDatabase(createDatabase);
82  		
83  		String serverName = (String) props
84  				.remove(DataSourceFactory.JDBC_SERVER_NAME);
85  		ds.setServerName(serverName);
86  		String portNumber = (String) props
87  				.remove(DataSourceFactory.JDBC_PORT_NUMBER);
88  		if (portNumber != null) {
89  			ds.setPortNumber(Integer.parseInt(portNumber));
90  		} else {
91  			ds.setPortNumber(1527);
92  		}
93  	}
94  
95  	private void doStartServer(Properties properties) {
96  		String host = (String) properties
97  				.get(DataSourceFactory.JDBC_SERVER_NAME);
98  		if (host == null) {
99  			host = "localhost";
100 		}
101 		String portNumberS = (String) properties
102 				.get(DataSourceFactory.JDBC_PORT_NUMBER);
103 		int portNumber = portNumberS == null ? 1527 : Integer
104 				.parseInt(portNumberS);
105 		boolean alreadyStarted = false;
106 		if (Activator.getInstance().getStartedServers().containsKey(host)) {
107 			alreadyStarted = Activator.getInstance().getStartedServers()
108 					.get(host).contains(portNumber);
109 		}
110 		if (!alreadyStarted) {
111 			try {
112 				InetAddress adress = InetAddress.getByName(host);
113 				NetworkServerControl control = new NetworkServerControl(adress, portNumber);
114 				String writer = (String) properties
115 						.remove(ClientConnectionConstant.LOG_FILE);
116 				if (writer == null) {
117 					writer = "derbyServer.log";
118 				}
119 				PrintWriter printWriter = new PrintWriter(writer);
120 				control.start(printWriter);
121 				Activator.getInstance().addNetworkControl(host, portNumber, control);
122 			} catch ( Exception e) {
123 				log.error("Error creating host adress", e);
124 			}
125 		}
126 
127 	}

129 	@Override
130 	public ConnectionPoolDataSource createConnectionPoolDataSource(
131 			Properties props) throws SQLException {
132 		ClientConnectionPoolDataSource40 ds = new ClientConnectionPoolDataSource40();
133 		setProperties(ds, props);
134 		
135 		return ds;
136 	}
137 
138 	@Override
139 	public XADataSource createXADataSource(Properties props)
140 			throws SQLException {
141 		ClientXADataSource40 ds = new ClientXADataSource40();
142 		setProperties(ds, props);
143 		return ds;
144 	}
145 
146 	@Override
147 	public Driver createDriver(Properties props) throws SQLException {
148 		ClientDriver40 driver = new ClientDriver40();
149 		
150 		return driver;
151 	}
